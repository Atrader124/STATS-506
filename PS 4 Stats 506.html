get_temp <- function(month, year, data, celsius = FALSE, average_fn = mean) {
  input_m <- month
  input_y <- year
  if (is.numeric(month)) {
    if(month < 1 || month > 12){
      print("Invalid month. Please provide a valid month (numeric 1-12 or a string).")
      return()
    }
    month <- as.numeric(month)
  } else if (is.character(month)) {
    input_m <- case_when(
    input_m %in% c("Jan", "January") ~ 1,
    input_m %in% c("Feb", "February") ~ 2,
    input_m %in% c("Mar", "March") ~ 3,
    input_m %in% c("Apr", "April") ~ 4,
    input_m %in% c("May") ~ 5,
    input_m %in% c("Jun", "June") ~ 6,
    input_m %in% c("Jul", "July") ~ 7,
    input_m %in% c("Aug", "August") ~ 8,
    input_m %in% c("Sep", "September") ~ 9,
    input_m %in% c("Oct", "October") ~ 10,
    input_m %in% c("Nov", "November") ~ 11,
    input_m %in% c("Dec", "December") ~ 12,
    TRUE ~ -1 # Default case
    )
  }
  
  filtered_data <- data %>%
    filter(month_numeric == input_m, year == input_y)
  
  if (nrow(filtered_data) == 0) {
    print("No data available for the specified month and year.")
    return()
  }

  mean_temperature <- filtered_data %>%
    summarize(temperature = average_fn(temp)) %>%
    pull(temperature)
  
  if (celsius) {
    mean_temperature <- (mean_temperature - 32) * 5/9
  }
  
  return(mean_temperature)
}

from recs2020

select sum(NWEIGHT) into :totalWeight from recs2020;
/* percentage of each state and sort */

select state_name,
total_weight / &totalWeight as percentage
from state_weights
